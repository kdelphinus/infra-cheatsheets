    вҳёпёҸ Kubernetes Cheat Sheet

    рҹҡҖ 1. мүҳ мһҗлҸҷмҷ„м„ұ м„Өм • (Shell Autocomplete)

    к°ҖмһҘ лЁјм Җ м„Өм •н•ҙм•ј н•  н•„мҲҳ кё°лҠҘмһ…лӢҲлӢӨ.

    bash

    source <(kubectl completion bash) # нҳ„мһ¬ м„ём…ҳм—җ м Ғмҡ©
echo "source <(kubectl completion bash)" >> ~/.bashrc # мҳҒкө¬ м Ғмҡ©
alias k=kubectl # 'k' лӢЁм¶•нӮӨ м„Өм •
complete -o default -F __start_kubectl k # 'k' лӢЁм¶•нӮӨм—җлҸ„ мһҗлҸҷмҷ„м„ұ м Ғмҡ©

    zsh

    source <(kubectl completion zsh)  # set up autocomplete in zsh into the current shell
echo '[[ $commands[kubectl] ]] && source <(kubectl completion zsh)' >> ~/.zshrc # add autocomplete permanently to your zsh shell
alias k=kubectl
complete -o default -F __start_kubectl k

    рҹ“Ӯ 2. нҒҙлҹ¬мҠӨн„° л°Ҹ м»Ён…ҚмҠӨнҠё кҙҖлҰ¬ (Config & Context)

    м—¬лҹ¬ нҒҙлҹ¬мҠӨн„°мҷҖ л„Өмһ„мҠӨнҺҳмқҙмҠӨлҘј кҙҖлҰ¬н•  л•Ң мӮ¬мҡ©н•©лӢҲлӢӨ.

    kubectl config view                        # м „мІҙ kubeconfig м„Өм • нҷ•мқё
kubectl config get-contexts                # м»Ён…ҚмҠӨнҠё лӘ©лЎқ нҷ•мқё
kubectl config current-context             # нҳ„мһ¬ м»Ён…ҚмҠӨнҠё нҷ•мқё
kubectl config use-context <name>          # м»Ён…ҚмҠӨнҠё ліҖкІҪ
kubectl config set-context --current --namespace=<ns>  # кё°ліё л„Өмһ„мҠӨнҺҳмқҙмҠӨ ліҖкІҪ (л§ӨлІҲ -n мһ…л Ҙ мғқлһө)
kubectl api-resources                      # мӮ¬мҡ© к°ҖлҠҘн•ң лҰ¬мҶҢмҠӨ мў…лҘҳ л°Ҹ лӢЁм¶•м–ҙ(Shortnames) нҷ•мқё
kubectl api-versions                       # мӮ¬мҡ© к°ҖлҠҘн•ң API к·ёлЈ№ лІ„м „ нҷ•мқё

    рҹӣ  3. лҰ¬мҶҢмҠӨ мғқм„ұ (Create & Dry Run)

    Best Practice: YAML нҢҢмқјмқ„ м§Ғм ‘ м§ңм§Җ л§җкі , лӘ…л №м–ҙлЎң лјҲлҢҖлҘј л§Ңл“  нӣ„ мҲҳм •н•ҳм„ёмҡ”. (--dry-run=client -o yaml)

    лҰ¬мҶҢмҠӨ лӘ…л №м–ҙ н…ңн”ҢлҰҝ
    Pod k run nginx --image=nginx --restart=Never --dry-run=client -o yaml > pod.yaml
    Deployment k create deploy web --image=nginx --replicas=3 --dry-run=client -o yaml > deploy.yaml
    Service k expose deploy web --port=80 --target-port=8080 --type=NodePort --dry-run=client -o yaml > svc.yaml
    CronJob k create cronjob my-job --image=busybox --schedule="*/1 * * * *" --dry-run=client -o yaml > cron.yaml
    Job k create job my-job --image=busybox --dry-run=client -o yaml > job.yaml
    Secret k create secret generic my-pass --from-literal=pwd=123 --dry-run=client -o yaml > sec.yaml
    ConfigMap k create cm my-config --from-file=config.txt --dry-run=client -o yaml > cm.yaml

    рҹ”Қ 4. мЎ°нҡҢ л°Ҹ н•„н„°л§Ғ (Get & Inspect)

    кё°ліё мЎ°нҡҢ

    kubectl get all -n <namespace>   # нҠ№м • л„Өмһ„мҠӨнҺҳмқҙмҠӨ м „мІҙ лҰ¬мҶҢмҠӨ мЎ°нҡҢ
kubectl get po,svc,deploy        # мЈјмҡ” лҰ¬мҶҢмҠӨл§Ң кіЁлқјм„ң мЎ°нҡҢ
kubectl get po -A                # лӘЁл“  л„Өмһ„мҠӨнҺҳмқҙмҠӨмқҳ Pod мЎ°нҡҢ
kubectl get nodes -o wide        # л…ёл“ң OS, м»Өл„җ лІ„м „, лӮҙл¶Җ IP нҷ•мқё
kubectl get events --sort-by=.metadata.creationTimestamp # мӢңк°„мҲң мқҙлІӨнҠё мЎ°нҡҢ (л””лІ„к№…мҡ©)

    м¶ңл Ҙ нҳ•мӢқ (Formatting) & м •л ¬

    kubectl get po -o wide                       # IP, Node м •ліҙ нҸ¬н•Ё
kubectl get po --show-labels                 # лқјлІЁ м •ліҙ нҸ¬н•Ё
kubectl get po -l app=nginx                  # нҠ№м • лқјлІЁ н•„н„°л§Ғ
kubectl get po --sort-by=.status.startTime   # мӢӨн–ү мӢңк°„мҲң м •л ¬
kubectl get po -o yaml > backup.yaml         # YAMLлЎң м „мІҙ мҠӨнҺҷ м¶”м¶ң

    JSONPath (лҚ°мқҙн„° м¶”м¶ң)

    # лӘЁл“  л…ёл“ңмқҳ мқҙлҰ„л§Ң м¶”м¶ң
kubectl get nodes -o jsonpath='{.items[*].metadata.name}'
# нҠ№м • Podмқҳ нҷҳкІҪліҖмҲҳ нҷ•мқё
kubectl get po <pod> -o jsonpath='{.spec.containers[0].env}'

    рҹ©ә 5. л””лІ„к№… л°Ҹ лЎңк·ё (Troubleshoot & Logs)

    лЎңк·ё л¶„м„қ

    kubectl logs <pod>                           # лЎңк·ё нҷ•мқё
kubectl logs -f <pod>                        # мӢӨмӢңк°„ лЎңк·ё (tail -f)
kubectl logs <pod> -c <container>            # нҠ№м • м»Ён…Ңмқҙл„Ҳ лЎңк·ё (Multi-container)
kubectl logs <pod> --previous                # мқҙм „ м»Ён…Ңмқҙл„Ҳ(CrashлӮң м§Ғнӣ„) лЎңк·ё нҷ•мқё
kubectl logs -l app=backend --all-containers=true # н•ҙлӢ№ лқјлІЁмқ„ к°Җм§„ лӘЁл“  Pod лЎңк·ё

    мғҒнғң м§„лӢЁ

    kubectl describe pod <pod>                   # Pod мғҒм„ё мғҒнғң л°Ҹ мқҙлІӨнҠё(Events) нҷ•мқё (вҳ…н•„мҲҳ)
kubectl describe node <node>                 # л…ёл“ң лҰ¬мҶҢмҠӨ л¶ҖмЎұ/мғҒнғң нҷ•мқё
kubectl top pods                             # Podлі„ CPU/Memory мӮ¬мҡ©лҹү (Metrics Server н•„мҡ”)
kubectl top nodes                            # л…ёл“ңлі„ лҰ¬мҶҢмҠӨ мӮ¬мҡ©лҹү

    м ‘мҶҚ л°Ҹ мӢӨн–ү

    kubectl exec -it <pod> -- /bin/bash          # м»Ён…Ңмқҙл„Ҳ мүҳ м ‘мҶҚ
kubectl exec -it <pod> -- ls -F /app         # м ‘мҶҚ м—Ҷмқҙ лӘ…л №м–ҙл§Ң мӢӨн–ү
kubectl cp <pod>:/path/file ./local_file     # нҢҢмқј ліөмӮ¬ (Pod -> Local)
kubectl cp ./local_file <pod>:/path/file     # нҢҢмқј ліөмӮ¬ (Local -> Pod)

    мһ„мӢң л””лІ„к№… Pod мӢӨн–ү

    # л„ӨнҠёмӣҢнҒ¬ н…ҢмҠӨнҠёмҡ©(curl, nslookup л“ұ) мһ„мӢң Pod мӢӨн–ү нӣ„ мӮӯм ң
kubectl run -it --rm debug --image=busybox --restart=Never -- sh

    рҹ”„ 6. м•ұ кҙҖлҰ¬ л°Ҹ мҠӨмјҖмқјл§Ғ (Deploy & Scale)

    м—…лҚ°мқҙнҠё л°Ҹ лЎӨл°ұ

    kubectl set image deploy/<name> nginx=nginx:1.19 # мқҙлҜём§Җ лІ„м „ м—…лҚ°мқҙнҠё (лЎӨл§Ғ м—…лҚ°мқҙнҠё)
kubectl rollout status deploy/<name>         # л°°нҸ¬ м§„н–ү мғҒнғң нҷ•мқё
kubectl rollout history deploy/<name>        # л°°нҸ¬ мқҙл Ҙ(Revision) нҷ•мқё
kubectl rollout undo deploy/<name>           # л°”лЎң мқҙм „ лІ„м „мңјлЎң лЎӨл°ұ
kubectl rollout undo deploy/<name> --to-revision=2 # нҠ№м • лІ„м „мңјлЎң лЎӨл°ұ
kubectl rollout pause deploy/<name>          # л°°нҸ¬ мқјмӢң м •м§Җ
kubectl rollout resume deploy/<name>         # л°°нҸ¬ мһ¬к°ң

    мҠӨмјҖмқјл§Ғ (Scaling)

    kubectl scale deploy/<name> --replicas=5     # мҲҳлҸҷ мҠӨмјҖмқјл§Ғ
kubectl autoscale deploy/<name> --min=2 --max=10 --cpu-percent=80 # HPA(Auto Scaling) м„Өм •

    рҹҢҗ 7. л„ӨнҠёмӣҢнҒ¬ (Networking)

    kubectl get svc -o wide                      # м„ңл№„мҠӨ ClusterIP, Selector нҷ•мқё
kubectl get ing                              # мқёк·ёл ҲмҠӨ(Ingress) мЈјмҶҢ нҷ•мқё
kubectl get endpoints <svc-name>             # м„ңл№„мҠӨмҷҖ м—°кІ°лҗң мӢӨм ң Pod IP(Endpoint) нҷ•мқё
kubectl port-forward <pod> 8080:80           # лЎңм»¬ нҸ¬нҠё нҸ¬мӣҢл”© (Pod)
kubectl port-forward svc/<name> 8080:80      # лЎңм»¬ нҸ¬нҠё нҸ¬мӣҢл”© (Service)

    рҹ—‘ 8. мӮӯм ң л°Ҹ м •лҰ¬ (Delete & Clean up)

    kubectl delete -f filename.yaml              # нҢҢмқј кё°мӨҖ мӮӯм ң
kubectl delete po <pod>                      # Pod мӮӯм ң
kubectl delete po -l app=nginx               # лқјлІЁ кё°мӨҖ мқјкҙ„ мӮӯм ң
kubectl delete ns <name>                     # л„Өмһ„мҠӨнҺҳмқҙмҠӨ мӮӯм ң (лӮҙл¶Җ лҰ¬мҶҢмҠӨ м „мІҙ мӮӯм ңлҗЁ)
# [к°•м ң мӮӯм ң] мў…лЈҢ(Terminating) мғҒнғңм—җм„ң л©Ҳм¶ҳ Pod мӮӯм ң
kubectl delete pod <pod> --grace-period=0 --force

    рҹҸ— 9. л…ёл“ң мң м§ҖліҙмҲҳ (Node Maintenance) - кҙҖлҰ¬мһҗмҡ©

    kubectl cordon <node>                        # мҠӨмјҖмӨ„л§Ғ мӨ‘лӢЁ (мӢ к·ң Pod л°°м№ҳ кёҲм§Җ)
kubectl uncordon <node>                      # мҠӨмјҖмӨ„л§Ғ мһ¬к°ң
kubectl drain <node> --ignore-daemonsets     # л…ёл“ң л№„мҡ°кё° (кё°мЎҙ PodлҘј лӢӨлҘё л…ёл“ңлЎң мқҙлҸҷ)
kubectl taint nodes <node> key=value:NoSchedule # н…ҢмқёнҠё м„Өм • (нҠ№м • Podл§Ң мҳӨкІҢ н•Ё)

    рҹ”җ 10. ліҙм•Ҳ, к¶Ңн•ң л°Ҹ нҶ нҒ° кҙҖлҰ¬ (Security, Auth & Tokens)

    лӮҙл¶Җ к¶Ңн•ң нҷ•мқё(RBAC)кіј л§ҢлЈҢлҗң нҒҙлқјмҡ°л“ң(CSP) мқёмҰқ нҶ нҒ°мқ„ к°ұмӢ н•ҳлҠ” н•„мҲҳ лӘ…л №м–ҙмһ…лӢҲлӢӨ.

    рҹ‘ЁвҖҚрҹ‘©вҖҚрҹ‘§вҖҚрҹ‘Ұ м—ӯн•  нҷ•мқё

    kubectl get serviceaccounts                  # м„ңл№„мҠӨ м–ҙм№ҙмҡҙнҠё лӘ©лЎқ
kubectl get clusterroles                     # нҒҙлҹ¬мҠӨн„° к¶Ңн•ң лӘ©лЎқ

    вҳҒпёҸ нҒҙлқјмҡ°л“ң(CSP) мқёмҰқ нҶ нҒ° к°ұмӢ  (Token Expired н•ҙкІ°)

    "You must be logged in to the server" м—җлҹ¬к°Җ лң° л•Ң, к°Ғ нҒҙлқјмҡ°л“ң CLIлЎң kubeconfigлҘј к°ұмӢ н•©лӢҲлӢӨ.

    # AWS EKS (Amazon)
aws eks update-kubeconfig --name <cluster-name> --region <region>

# GCP GKE (Google)
gcloud container clusters get-credentials <cluster-name> --region <region>

# Azure AKS (Microsoft)
az aks get-credentials --resource-group <rg-name> --name <cluster-name>

    рҹ”‘ м„ңл№„мҠӨ м–ҙм№ҙмҡҙнҠё нҶ нҒ° кҙҖлҰ¬ (Service Account)

    CI/CD м—°лҸҷмқҙлӮҳ лҢҖмӢңліҙл“ң лЎңк·ёмқёмҡ© нҶ нҒ°мқ„ л°ңкёүн•©лӢҲлӢӨ. (K8s v1.24+ н•„мҲҳ)

    # [Modern] нҶ нҒ° мҰүмӢң л°ңкёү (мң нҡЁкё°к°„ м„Өм • к°ҖлҠҘ)
kubectl create token <service-account-name> --duration=24h

# [Legacy] кё°мЎҙ Secret кё°л°ҳ нҶ нҒ° м¶”м¶ң (Base64 л””мҪ”л”©)
kubectl get secret <secret-name> -o jsonpath='{.data.token}' | base64 --decode

    рҹӣЎпёҸ к¶Ңн•ң нҷ•мқё (RBAC Check)

    нҳ„мһ¬ мӮ¬мҡ©мһҗлӮҳ нҠ№м • кі„м •мқҙ мһ‘м—…мқ„ мҲҳн–үн•  к¶Ңн•ңмқҙ мһҲлҠ”м§Җ нҷ•мқён•©лӢҲлӢӨ.

    # "лӮҙк°Җ л°°нҸ¬(Deployment)лҘј л§Ңл“Ө мҲҳ мһҲлӮҳ?" (Yes/No л°ҳнҷҳ)
kubectl auth can-i create deployments

# нҠ№м • л„Өмһ„мҠӨнҺҳмқҙмҠӨм—җм„ң к¶Ңн•ң нҷ•мқё
kubectl auth can-i delete pods -n <namespace>

# [кҙҖлҰ¬мһҗмҡ©] нҠ№м • м„ңл№„мҠӨ м–ҙм№ҙмҡҙнҠё(sa)к°Җ к¶Ңн•ңмқҙ мһҲлҠ”м§Җ нҷ•мқё (impersonate)
kubectl auth can-i get nodes --as=system:serviceaccount:<ns>:<sa-name>

    рҹҺ« нҒҙлҹ¬мҠӨн„° мЎ°мқё нҶ нҒ° (Kubeadm м „мҡ©)

    мҳЁн”„л ҲлҜёмҠӨ(On-premise) нҷҳкІҪм—җм„ң л…ёл“ңлҘј м¶”к°Җн•  л•Ң мӮ¬мҡ©н•©лӢҲлӢӨ.

    # мң нҡЁн•ң мЎ°мқё нҶ нҒ° лӘ©лЎқ нҷ•мқё
kubeadm token list

# нҶ нҒ°мқҙ л§ҢлЈҢлҗҳм—Ҳмқ„ л•Ң мһ¬мғқм„ұ (мЎ°мқё лӘ…л №м–ҙ м „мІҙ м¶ңл Ҙ)
kubeadm token create --print-join-command

    рҹ’Ў нҢҒ (Tips)

    • лҰ¬мҶҢмҠӨк°Җ м•Ҳ м§ҖмӣҢм§Ҳ л•Ң: finalizers м„Өм • л•Ңл¬ёмқј мҲҳ мһҲмҠөлӢҲлӢӨ. kubectl editмңјлЎң finalizers н•ӯлӘ©мқ„ м§Җмҡ°л©ҙ мӮӯм ңлҗ©лӢҲлӢӨ.
    • YAML м¶”м¶ң мҠөкҙҖ: мҡҙмҳҒ мӨ‘мқё лҰ¬мҶҢмҠӨлҘј мҲҳм •н•ҳкё° м „м—җ л°ҳл“ңмӢң kubectl get <resource> -o yaml > backup.yamlлЎң л°ұм—…н•ҳм„ёмҡ”.
    • Watch лӘЁл“ң: л°°нҸ¬ нӣ„ kubectl get pods -wлҘј мјңл‘җл©ҙ мғҒнғң ліҖнҷ”лҘј мӢӨмӢңк°„мңјлЎң ліј мҲҳ мһҲм–ҙ лӢөлӢөн•Ёмқҙ мӨ„м–ҙл“ӯлӢҲлӢӨ.

    вҡ“ Helm Essential Cheat Sheet (Bonus)

    Kubernetesмқҳ aptлӮҳ yum к°ҷмқҖ нҢЁнӮӨм§Җ л§ӨлӢҲм Җ, Helmмқҳ н•„мҲҳ лӘ…л №м–ҙмһ…лӢҲлӢӨ.

    1. м ҖмһҘмҶҢ(Repo) кҙҖлҰ¬

    м°ЁнҠё(нҢЁнӮӨм§Җ)лҘј лӢӨмҡҙлЎңл“ңн•  м ҖмһҘмҶҢлҘј кҙҖлҰ¬н•©лӢҲлӢӨ.

    helm repo add <name> <url>       # м ҖмһҘмҶҢ м¶”к°Җ (мҳҲ: bitnami https://charts.bitnami.com/bitnami)
helm repo update                 # м ҖмһҘмҶҢ мөңмӢ  м •ліҙ к°ұмӢ  (apt updateмҷҖ лҸҷмқј)
helm repo list                   # л“ұлЎқлҗң м ҖмһҘмҶҢ лӘ©лЎқ нҷ•мқё
helm search repo <keyword>       # м°ЁнҠё кІҖмғү (мҳҲ: helm search repo nginx)

    2. м„Өм№ҳ л°Ҹ кҙҖлҰ¬ (Install & Manage)

    м• н”ҢлҰ¬мјҖмқҙм…ҳ(Release)мқ„ м„Өм№ҳн•ҳкі  м—…к·ёл Ҳмқҙл“ңн•©лӢҲлӢӨ.

    # м°ЁнҠё м„Өм№ҳ (кё°ліё)
helm install <release-name> <repo/chart>
# мҳҲ: helm install my-nginx bitnami/nginx

# нҠ№м • л„Өмһ„мҠӨнҺҳмқҙмҠӨм—җ м„Өм№ҳн•ҳл©° л„Өмһ„мҠӨнҺҳмқҙмҠӨ мһҗлҸҷ мғқм„ұ
helm install <release-name> <repo/chart> -n <ns> --create-namespace

# м„Өм • к°’(values.yaml)мқ„ ліҖкІҪн•ҳл©° м„Өм№ҳ
helm install <release-name> <repo/chart> --set replicaCount=3

# мӮ¬мҡ©мһҗ м •мқҳ values нҢҢмқјлЎң м„Өм№ҳ
helm install <release-name> <repo/chart> -f values.yaml

# л°°нҸ¬лҗң м•ұ лӘ©лЎқ нҷ•мқё
helm list -A  # лӘЁл“  л„Өмһ„мҠӨнҺҳмқҙмҠӨ мЎ°нҡҢ

    3. м—…к·ёл Ҳмқҙл“ң л°Ҹ мӮӯм ң (Upgrade & Uninstall)

    # м°ЁнҠё лІ„м „ м—…к·ёл Ҳмқҙл“ң лҳҗлҠ” м„Өм • ліҖкІҪ
helm upgrade <release-name> <repo/chart> -f new-values.yaml

# л°°нҸ¬лҗң м•ұ мӮӯм ң
helm uninstall <release-name>

    4. л””лІ„к№… л°Ҹ кІҖмҰқ (Debug & Verify)

    Pro Tip: helm templateмқҖ мӢӨм ңлЎң нҒҙлҹ¬мҠӨн„°м—җ м„Өм№ҳн•ҳм§Җ м•Ҡкі , мғқм„ұлҗ  YAML нҢҢмқјл§Ң лҜёлҰ¬ ліҙм—¬мӨҚлӢҲлӢӨ. (--dry-runкіј мң мӮ¬)

    # л ҢлҚ”л§Ғлҗ  YAML лҜёлҰ¬ліҙкё° (м„Өм№ҳ м „ кІҖмҰқ н•„мҲҳ лӢЁкі„)
helm template <release-name> <repo/chart> -f values.yaml > result.yaml

# нҳ„мһ¬ мӢӨн–ү мӨ‘мқё лҰҙлҰ¬мҰҲмқҳ м Ғмҡ©лҗң values к°’ нҷ•мқё
helm get values <release-name>

# нҳ„мһ¬ мӢӨн–ү мӨ‘мқё лҰҙлҰ¬мҰҲмқҳ м „мІҙ Manifest нҷ•мқё
helm get manifest <release-name>

# лҰҙлҰ¬мҰҲмқҳ л°°нҸ¬ мқҙл Ҙ л°Ҹ мғҒнғң нҷ•мқё
helm status <release-name>
